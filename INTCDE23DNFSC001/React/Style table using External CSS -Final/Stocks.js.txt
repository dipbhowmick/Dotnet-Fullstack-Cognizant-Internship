class Stocks extends Component {
	//IMPLEMENT YOUR CODE HERE
	constructor(props){
	    super(props);
	    this.state = { stocks: data};
	}
		render(){

	          <>

}

class Display extends Component {
	  //IMPLEMENT YOUR CODE HERE 
	render(){

}
